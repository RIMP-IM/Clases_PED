void ListaSimple::insFinal(int dato) {
	nodo* p, * q;

	nodo* nuevo = new nodo;
	nuevo->dato = dato;
	nuevo->sig = NULL;
	if (pInicio == NULL)
		pInicio = nuevo;
	else {
		p = pInicio;
		q = NULL;

		while (p != NULL)
		{
			q = p;
			p = p->sig;
		}
		q->sig = nuevo;
	}
}

	p = pInicio;
	q = NULL;
	while (p != NULL && p->dato != datoRef)
	{
		q = p;
		p = p->sig;
	}

	if (q == NULL)
	{
		if (p == NULL) {
			cout << "Insertaste el primero" << endl;
			pInicio = nuevo;
			pInicio->sig = NULL;
		}
		else
		{
			cout << "Dato de referencia esta al principio" << endl;
			pInicio = nuevo;
			pInicio->sig = p;
		}
	}
	else {
		if (p == NULL) {
			cout << "Dato de referencia no encontrado" << endl;
			q->sig = nuevo;
			nuevo->sig = NULL;
		}
		else
		{
			cout << "Dato insertado a la mitad de la lista" << endl;
			q->sig = nuevo;
			nuevo->sig = p;
		}
	}
}
