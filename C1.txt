void llenarArreglo(int* p, int n) {
	if (n >= 0) {
		cin >> *p;
		llenarArreglo(p + 1, n - 1);
	}
}

void mostrarArreglo(int* p, int n) {
	if (n >= 0) {
		cout << *p << " ";
		mostrarArreglo(p + 1, n - 1);
	}
}
void invertirArreglo(int* p, int* q) {
	int temp;
	if (p < q) {
		temp = *p;
		*p = *q;
		*q = temp;
		invertirArreglo(p + 1, q-1);
	}
}


void ullman(int Numero) {
